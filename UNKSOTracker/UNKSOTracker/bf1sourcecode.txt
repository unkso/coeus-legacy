    function updateSettings(graph) {   
        if (window.localStorage != null) {
            window.localStorage.setItem('overviewgraph', graph);
        }
    }

    $(function () {
        if (window.localStorage != null) {
            if (window.localStorage.getItem('overviewgraph')) {
                if (window.localStorage.getItem('overviewgraph') == 'win') {
                    win();
                    return;
                } else if (window.localStorage.getItem('overviewgraph') == 'kd') {
                    kd();
                    return;
                } else if (window.localStorage.getItem('overviewgraph') == 'ss') {
                    ss();
                    return;
                } else if (window.localStorage.getItem('overviewgraph') == 'score') {
                    score();
                    return;
                }

            }
        }
        score();
    });
    function ss() {
        updateSettings('ss');

        var chart = $('#chartkd');

        chart.html('<span class="fa fa-spin fa-cog fa-4x"></span>');
        $.ajax({
            url: '/bf1/partials/_chartoverview',
            data: { playerId: '1484565', column: 'ss' },
            type: 'GET'
        })
        .done(function (data) {
            if (data.categories.length == 0) {
                $('#chartkd')
                    .html('Player has not played in the last month')
                    .attr('style', 'height: 20px;');

                return;
            }
            var chart;
            var chartingOptions = {
                rangeSelector: {
                    enabled: false
                },
                chart: {
                    borderRadius: 0,
                    backgroundColor: 'rgba(0, 0, 0, 0)',
                    type: 'line'
                },
                title: {
                    text: 'BTR Score by Day',
                    style: {
                        color: '#80D9FF'
                    }
                },
                xAxis: {
                    categories: data.categories,
                    labels: {
                        formatter: function () {
                            if (this.isFirst && this.value == 'Jan 01') {
                                return 'Prior Data';
                            }

                            return this.value;
                        }
                    }
                },
                yAxis: [{
                    title: {
                        text: 'Daily Up/Down',
                        style: {
                            color: Highcharts.getOptions().colors[1]
                        }
                    }
                }, {
                    title: {
                        text: 'BTR Score',
                        style: {
                            color: Highcharts.getOptions().colors[0]
                        }
                    },
                    opposite: true
                }],
                plotOptions: {
                    line: {
                        enableMouseTracking: true
                    }
                },
                tooltip: {
                    positioner: function (labelWidth, labelHeight, point) {
                        var tooltipX, tooltipY;
                        if (point.plotX + labelWidth > chart.plotWidth) {
                            tooltipX = point.plotX + chart.plotLeft - labelWidth - 150;
                        } else {
                            tooltipX = point.plotX + chart.plotLeft - 150;
                        }
                        tooltipY = point.plotY + chart.plotTop - 20;
                        return {
                            x: tooltipX,
                            y: tooltipY
                        };
                    }
                },
                series: [
                    { name: 'Score Up/Down', type: 'column', data: data.ssbyday },
                    { name: 'Overall Site Score', type: 'spline', data: data.ss, yAxis: 1 }
                ]
            };
            chart = $('#chartkd').highcharts(chartingOptions).highcharts();
        })
        .error(function () {
            $('#chartkd')
                    .html('Error loading data. Please check again later.')
                    .attr('style', 'height: 20px;');
        });
    }
