    function updateSettings(graph) {
        if (window.localStorage != null) {
            window.localStorage.setItem('overviewgraph', graph);
        }
    }

    $(function () {
        if (window.localStorage != null) {
            if (window.localStorage.getItem('overviewgraph')) {
                if (window.localStorage.getItem('overviewgraph') == 'win') {
                    win();
                    return;
                } else if (window.localStorage.getItem('overviewgraph') == 'kd') {
                    kd();
                    return;
                } else if (window.localStorage.getItem('overviewgraph') == 'killsdeaths') {
                    killsdeaths();
                    return;
                } else if (window.localStorage.getItem('overviewgraph') == 'accuracy') {
                    accuracy();
                    return;
                } else if (window.localStorage.getItem('overviewgraph') == 'headshots') {
                    headshots();
                    return;
                }
            }
        }
        kd();
    });

    function kd() {
        updateSettings('kd');

        var chart = $('#chartkd');

        chart.html('<span class="fa fa-spin fa-cog fa-4x"></span>');
        $.ajax({
            url: '/r6/partials/_chartoverview',
            data: { playerId: '308bb5a2-be5e-47c0-9698-5fb4338cdfef', column: 'kd' },
            type: 'GET'
        })
        .done(function (data) {
            if (data.categories.length == 0) {
                $('#chartkd')
                    .html('We do not have any history for this player, go play!')
                    .attr('style', 'height: 20px;');

                return;
            }

            var chart;
            var chartingOptions = {
                rangeSelector: {
                    enabled: false
                },
                chart: {
                    borderRadius: 0,
                    backgroundColor: 'rgba(0, 0, 0, 0)',
                    type: 'line'
                },
                title: {
                    text: 'Player K/d By Day',
                    style: {
                        color: '#80D9FF'
                    }
                },
                xAxis: {
                    categories: data.categories,
                    labels: {
                        formatter: function () {
                            if (this.isFirst && this.value == 'Jan 01') {
                                return 'Prior Data';
                            }

                            return this.value;
                        }
                    }
                },
                yAxis: [{
                    title: {
                        text: 'Daily Kd',
                        style: {
                            color: Highcharts.getOptions().colors[1]
                        }
                    }
                }, {
                    title: {
                        text: 'Overall Kd',
                        style: {
                            color: Highcharts.getOptions().colors[0]
                        }
                    },
                    opposite: true
                }],
                tooltip: {
                    positioner: function (labelWidth, labelHeight, point) {
                        var tooltipX, tooltipY;
                        if (point.plotX + labelWidth > chart.plotWidth) {
                            tooltipX = point.plotX + chart.plotLeft - labelWidth - 100;
                        } else {
                            tooltipX = point.plotX + chart.plotLeft - 100;
                        }
                        tooltipY = point.plotY + chart.plotTop - 20;
                        return {
                            x: tooltipX,
                            y: tooltipY
                        };
                    }
                },
                plotOptions: {
                    line: {
                        enableMouseTracking: true
                    }
                },
                series: [{ name: 'K/d Overall', data: data.kd }, { name: 'K/d Daily', data: data.kdbyday }]
            };

            chart = $('#chartkd').highcharts(chartingOptions).highcharts();
        })
        .error(function () {
            $('#chartkd')
                    .html('Error loading data. Please check again later.')
                    .attr('style', 'height: 20px;');
        });
    }
